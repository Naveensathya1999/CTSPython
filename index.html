11. Rythm Composer

def find_prime(start, end):
    primes = []
    for num in range(start, end+1):
        if num==1:
            continue
        for i in range(2,num//2+1):
            if(num%i==0):
                break
        else:
            primes.append(num)
    return primes

start = int(input())
end = int(input())

if start<0 or end<0 or start>end:
    print("Invalid range")
elif start==end or (start==0 and end==1):
    print("There is no prime numbers in this range")
else:
    primes = find_prime(start, end)
    if len(primes)!=0:
        for prime in primes:
            print(prime,end=' ')
    else:
        print("There is no prime numbers in this range")

Module date.py

import calendar

def is_leap(year):
    return calendar.isleap(year)

def month_name(no):
    return calendar.month_name[no]

def days_in_month(month, year):
    return calendar.monthrange(year, month)[1]
